#************* Módulo 08 - Mudança Estrutural (Alternativa ao teste de Chow)  *************#

# Nota:
#      ano =	Período						
#        Y =	poupança						
#        X =	renda	
#        d =  dummy
#       dx = Dt*Xt

# Prepararando base de dados
base <- read.csv2("Tab_8-9_chow_test.txt", sep = "\t")
str(base)
head(base)

# Teste de Chow
install.packages("strucchange")                     # pacote para realizar teste de estabilidade estrutural
library("strucchange")
sctest(y~x, type = "Chow", data = base, point = 12) # se p-value for maior que o nível de significância, então falha em rejeitar H0, logo não há mudança estrutural

# Alternativa ao teste de Chow
FRA <- lm(y~x+d+dx, data = base)
summary(FRA)
